  console.warn(`${key} is null, pls check!`);

    # 标记语句灵活跳出嵌套循环

    for (let i = 0; i < 3; i++) {
  for (let j = 0; j < 3; j++) {
    if (i === j) {
      continue firstLoop; // 继续 firstLoop 循环
      // break firstLoop; // 中止 firstLoop 循环
    }
    console.log(`i = ${i}, j = ${j}`);
  }
}
// 输出
(i = 1), (j = 0);
(i = 2), (j = 0);
(i = 2), (j = 1);

for (let i = 0; i < 3; i++) {
  for (let j = 0; j < 3; j++) {
    if (i === j) {
      continue;
    }
    console.log(`i = ${i}, j = ${j}`);
  }
}
// 输出
(i = 0), (j = 1);
(i = 0), (j = 2);
(i = 1), (j = 0);
(i = 1), (j = 2);
(i = 2), (j = 0);
(i = 2), (j = 1);

//firstLoop是关键字
