        1. form 表单触发下载

          这里不展开，需要用 post 请求，适用于前端上传信息然后下载

        TIP

        Content-Disposition：[inline|attachment|attachment; filename="filename.jpg"] Content-Disposition 响应头指示回复的内容该以何种形式展示
        inline 按照浏览器自己的处理方式是直接下载还是预览
        attachment 设置弹出下载
        attachment; filename="filename.jpg 设置弹出下载并 filename.jpg 命名

        # 文件流下载

        文件流下载 excel🌰

        function exportFile(url, data) {
  request({
    //告诉服务器需要返回一个buffer，不然下载下来很可能是乱码
    responseType: "arraybuffer",
    url: url,
    method: "post",
    data: data,
  }).then((res) => {
    const content = res.data
    //转换成blob对象
    const blob = new Blob([content], {
      type: "application/vnd.ms-excel;charset=utf-8", //不同文件类型可能不同
    })
    //获取文件名字
    const fileName = res.headers["content-disposition"].split("=")[1]

    const elink = document.createElement("a")
    elink.download = decodeURI(fileName)
    elink.style.display = "none"
    console.log(res)
    //生成blobUrl
    elink.href = URL.createObjectURL(blob)
    document.body.appendChild(elink)
    elink.click()

    URL.revokeObjectURL(elink.href) // 释放URL 对象
    document.body.removeChild(elink)
  })
}

        原生小黄人下载

        function request() {
  const req = new XMLHttpRequest()
  req.open("POST", "<接口地址>", true)
  req.responseType = "blob"
  req.setRequestHeader("Content-Type", "application/json")
  req.onload = function() {
    const data = req.response
    const a = document.createElement("a")
    const blob = new Blob([data])
    const blobUrl = window.URL.createObjectURL(blob)
    download(blobUrl)
  }
  req.send("<请求参数：json字符串>")
}
function download(blobUrl) {
  const a = document.createElement("a")
  a.style.display = "none"
  a.download = "<文件名>"
  a.href = blobUrl
  a.click()
  document.body.removeChild(a)
}
